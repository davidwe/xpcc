00102         struct GpioMode {
00103             ALWAYS_INLINE static void setMode(uint32_t m) {
00104                 uint32_t tmp = reinterpret_cast<GPIO_TypeDef*>(P)->CRH;
00105                 tmp &= ~(0xf << ((N - 8) * 4));
00106                 tmp |= (m << ((N - 8) * 4));
00107                 reinterpret_cast<GPIO_TypeDef*>(P)->CRH = tmp;
00108             }
00109         };
