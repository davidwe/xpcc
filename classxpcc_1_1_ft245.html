Detailed Description

The FT245R USB FIFO from Future Technology Devices International is single chip USB to parallel FIFO bidirectional data transfer interface.

Example

 #include <xpcc/architecture.hpp>
 #include <xpcc/driver/connectivity/ft245.hpp>
 
 GPIO__PORT(FtPort, A);
 GPIO__OUTPUT(FtRd, B, 0);
 GPIO__OUTPUT(FtWr, B, 1);
 GPIO__INPUT(FtTxe, B, 2);
 GPIO__INPUT(FtRxf, B, 3);
 
 MAIN_FUNCTION
 {
     xpcc::Ft245<FtPort, FtRd, FtWr, FtRxf, FtTxe> ft245;
     
     uint8_t out = 0;
     while(1)
     {
          ft245.write(out++);
            
            char loop;
            if (ft245.read(loop)) {
              ft245.write(0xff ^ loop);
            }
     }
 }
